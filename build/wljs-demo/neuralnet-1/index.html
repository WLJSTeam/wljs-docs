  data = {{0.,0.}, {1.0,0.}},
  loss = 0.0

  updateGraph[net_, l_] := (
    data = Table[{t, net[t]}, {t,2,5.0,0.3}];
    loss = l;
  );

  Plot[f[x], {x,2,5}, Epilog->{
    Red, Line[data // Offload],
    Black, Text[loss // Offload, {3.2, -0.93}]
  }]
]
